CREATE TABLE user (
    email VARCHAR(255) NOT NULL,
    password VARCHAR(255) NOT NULL,
    PRIMARY KEY (email)
);

CREATE TABLE user_info (
    email VARCHAR(255) NOT NULL,
    name VARCHAR(255) NOT NULL,
    age INT,
    gender ENUM('男', '女', '其他') NOT NULL,
    university VARCHAR(255),
    major VARCHAR(255),
    phone VARCHAR(20),
    PRIMARY KEY (email),
    FOREIGN KEY (email) REFERENCES user(email)
);

CREATE TABLE user_profile (
    email VARCHAR(255) NOT NULL,            -- 用户邮箱，作为外键
    resume_filename VARCHAR(255),           -- 存储简历文件名
    photo_filename VARCHAR(255),            -- 存储照片文件名
    PRIMARY KEY (email),                    -- 将 email 作为主键
    FOREIGN KEY (email) REFERENCES user(email)  -- 外键约束，email关联user表中的email
);

CREATE TABLE admin (
    email VARCHAR(255) PRIMARY KEY,
    password VARCHAR(255) NOT NULL
);
//岗位信息
CREATE TABLE job_info (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,       -- 自增主键
    name VARCHAR(1000) NOT NULL,                 -- 岗位名称
    number BIGINT NOT NULL,                     -- 招聘人数
    demand TEXT,                                -- 岗位需求
    salary VARCHAR(100)                         -- 薪资待遇
);
//投递信息
CREATE TABLE submit_state (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    email VARCHAR(255) NOT NULL,
    job_state INT NOT NULL,
    job_id BIGINT,
    job_name VARCHAR(255),
    CONSTRAINT fk_job_id FOREIGN KEY (job_id) REFERENCES job_info(id)
);
